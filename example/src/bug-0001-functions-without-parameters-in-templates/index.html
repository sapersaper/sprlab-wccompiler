Expected Compiler Enhancement

Add validation in the parser to detect name collisions early:

// In lib/parser-extractors.js or lib/parser.js
function validateNameCollisions(signals, computeds, methods, props) {
  const allNames = new Set();
  
  // Check for collisions between signals and functions
  for (const signal of signals) {
    if (methods.some(m => m.name === signal.name)) {
      throw new Error(
        `Name collision: '${signal.name}' is defined as both a signal and a function.\\n` +
        `Solution: Use different names or convert to computed().`
      );
    }
    allNames.add(signal.name);
  }
  
  // Similar checks for computeds, props, etc.
}
